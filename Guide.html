01

Introduction to Riverpod

Riverpod is a reactive caching and data-binding framework for Flutter and Dart. Created by Remi Rousseau (the author of Provider), it solves all of Provider's limitations while being 100% compile-safe.

🛡️

Compile-Safe

No ProviderNotFoundException at runtime. Errors are caught at compile time.

🧪

Fully Testable

Override any provider in tests without special test widgets.

⚡

Auto-Dispose

Automatically destroys state when it's no longer needed.

🔗

Combinable

Providers can easily depend on each other without limitations.

🌊

Async-First

Built-in support for Future, Stream, and loading/error states.

🏗️

Code Generation

Less boilerplate with @riverpod annotation and build_runner.

Riverpod vs Provider vs BLoC

ConsumerWidget vs ConsumerStatefulWidget

DART 
// Preferred for most widgets
class MyWidget extends ConsumerWidget {
  const MyWidget({super.key});

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    // ref.watch() → rebuilds when value changes
    final count = ref.watch(counterProvider);

    // ref.read() → one-time read, no rebuild
    // ref.listen() → side effects on change

    return Text('Count: $count');
  }
}
DART 
// Use when you need initState / dispose
class MyWidget extends ConsumerStatefulWidget {
  const MyWidget({super.key});

  @override
  ConsumerState<MyWidget> createState() => _MyWidgetState();
}

class _MyWidgetState extends ConsumerState<MyWidget> {
  @override
  void initState() {
    super.initState();
    // ref is available here!
    ref.read(counterProvider);
  }

  @override
  Widget build(BuildContext context) {
    final count = ref.watch(counterProvider);
    return Text('Count: $count');
  }
}
DART 
// Requires hooks_riverpod package
import 'package:hooks_riverpod/hooks_riverpod.dart';
import 'package:flutter_hooks/flutter_hooks.dart';

class MyWidget extends HookConsumerWidget {
  const MyWidget({super.key});

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    // Mix hooks and riverpod
    final controller = useTextEditingController();
    final count = ref.watch(counterProvider);

    return Text('Count: $count');
  }
}

ref.watch vs ref.read vs ref.listen

MethodRebuilds widget?Use case
ref.watch(p)✅ YesDisplay value in build()
ref.read(p)❌ NoEvent handlers, initState
ref.listen(p, cb)❌ NoSide effects (navigation, snackbar)
ref.watch(p.notifier)❌ NoGet Notifier without subscribing to state
ref.invalidate(p)—Force refresh a provider
ref.refresh(p)—Invalidate + return new value
⚠️ Warning Never use ref.read() inside build(). It will not re-render when the value changes. Always use ref.watch() in build methods.
04

Provider

The simplest provider. Holds an immutable value that never changes. Perfect for dependency injection (services, repositories, configs).

DART 
// ── Traditional syntax ──────────────────
final apiBaseUrlProvider = Provider<String>((ref) {
  return 'https://api.example.com';
});

// Inject a service class
final httpClientProvider = Provider<Dio>((ref) {
  final baseUrl = ref.watch(apiBaseUrlProvider);
  return Dio(BaseOptions(baseUrl: baseUrl));
});

final userRepositoryProvider = Provider<UserRepository>((ref) {
  final dio = ref.watch(httpClientProvider);
  return UserRepository(dio);
});

// ── Code-gen syntax (@riverpod) ──────────
@riverpod
UserRepository userRepository(UserRepositoryRef ref) {
  final dio = ref.watch(httpClientProvider);
  return UserRepository(dio);
}
✅ Best Practice Use Provider for dependency injection of services, repositories, and configs. Prefer code-gen syntax for new code — it's less verbose and generates better types.
05

StateProvider

Exposes a piece of mutable state. Best for simple values: toggles, selected tab index, filter values. For complex state, use NotifierProvider.

DART 
// Define
final counterProvider = StateProvider<int>((ref) => 0);
final isDarkModeProvider = StateProvider<bool>((ref) => false);
final selectedTabProvider = StateProvider<int>((ref) => 0);

// Read in widget
class CounterPage extends ConsumerWidget {
  const CounterPage({super.key});

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final count = ref.watch(counterProvider);

    return Column(children: [
      Text('$count'),
      ElevatedButton(
        // Update using .notifier
        onPressed: () => ref.read(counterProvider.notifier).state++,
        child: const Text('Increment'),
      ),
      ElevatedButton(
        // Update using update()
        onPressed: () => ref.read(counterProvider.notifier)
            .update((state) => state - 1),
        child: const Text('Decrement'),
      ),
    ]);
  }
}
⚠️ Limitation Don't use StateProvider for complex objects like lists or maps. Mutating the object without replacing it won't trigger a rebuild. Use NotifierProvider instead.
06

NotifierProvider

The replacement for StateNotifierProvider (Riverpod 2.0+). Use for complex state logic. State is managed inside a Notifier class with named methods.

DART 
// ── Model ───────────────────────────────
class Todo {
  const Todo({required this.id, required this.title, this.completed = false});
  final String id;
  final String title;
  final bool completed;

  Todo copyWith({String? id, String? title, bool? completed}) => Todo(
    id: id ?? this.id,
    title: title ?? this.title,
    completed: completed ?? this.completed,
  );
}

// ── Notifier ────────────────────────────
class TodosNotifier extends Notifier<List<Todo>> {
  @override
  List<Todo> build() {
    // Initial state — can also watch other providers here
    return [];
  }

  void add(String title) {
    state = [
      ...state,
      Todo(id: DateTime.now().toString(), title: title),
    ];
  }

  void toggle(String id) {
    state = [
      for (final todo in state)
        if (todo.id == id) todo.copyWith(completed: !todo.completed)
        else todo,
    ];
  }

  void remove(String id) {
    state = state.where((t) => t.id != id).toList();
  }
}

// ── Provider ────────────────────────────
final todosProvider = NotifierProvider<TodosNotifier, List<Todo>>(
  TodosNotifier.new,
);

// ── With code generation ─────────────────
@riverpod
class Todos extends _$Todos {
  @override
  List<Todo> build() => [];

  void add(String title) { /* same as above */ }
  void toggle(String id) { /* same as above */ }
}
// Generated: todosProvider, TodosProvider

// ── Widget Usage ────────────────────────
class TodoList extends ConsumerWidget {
  const TodoList({super.key});

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final todos = ref.watch(todosProvider);

    return ListView.builder(
      itemCount: todos.length,
      itemBuilder: (ctx, i) {
        final todo = todos[i];
        return ListTile(
          title: Text(todo.title),
          leading: Checkbox(
            value: todo.completed,
            onChanged: (_) => ref.read(todosProvider.notifier).toggle(todo.id),
          ),
        );
      },
    );
  }
}
07

FutureProvider

For one-time async operations. Returns an AsyncValue<T> which represents loading, data, or error states.

DART 
// Define
final userProvider = FutureProvider<User>((ref) async {
  final repo = ref.watch(userRepositoryProvider);
  return repo.fetchCurrentUser();
});

// Widget — handling AsyncValue
class UserProfile extends ConsumerWidget {
  const UserProfile({super.key});

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final userAsync = ref.watch(userProvider);

    // Pattern 1: .when()
    return userAsync.when(
      loading: () => const CircularProgressIndicator(),
      error: (err, stack) => Text('Error: $err'),
      data: (user) => Text('Hello ${user.name}'),
    );

    // Pattern 2: .maybeWhen()
    return userAsync.maybeWhen(
      data: (user) => Text(user.name),
      orElse: () => const SizedBox(),
    );

    // Pattern 3: switch expression (Dart 3)
    return switch (userAsync) {
      AsyncData(:final value) => Text(value.name),
      AsyncError(:final error) => Text('$error'),
      _ => const CircularProgressIndicator(),
    };
  }
}

// Force refresh (e.g., pull-to-refresh)
void refresh(WidgetRef ref) {
  ref.invalidate(userProvider);
  // OR: await ref.refresh(userProvider.future);
}

// Show previous data while refreshing
final userAsync = ref.watch(userProvider);
if (userAsync.isLoading && userAsync.hasValue) {
  // Show old data with a loading indicator on top
  return Stack(children: [
    Text(userAsync.value!.name),
    const LinearProgressIndicator(),
  ]);
}

AsyncValue API

Property/MethodDescription
asyncValue.valueThe data (null if loading/error)
asyncValue.errorThe error (null if loading/data)
asyncValue.isLoadingTrue during initial load or refresh
asyncValue.hasValueTrue if data exists (even while refreshing)
asyncValue.hasErrorTrue if in error state
asyncValue.when(...)Handle all three states
asyncValue.whenData(fn)Transform data, propagate loading/error
asyncValue.valueOrNullData or null safely
AsyncValue.guard(fn)Wrap a Future, catching exceptions
08

StreamProvider

Listens to a Stream and exposes an AsyncValue<T>. Perfect for real-time data (Firestore, WebSockets, sensors).

DART 
// Firestore real-time example
final messagesProvider = StreamProvider<List<Message>>((ref) {
  return FirebaseFirestore.instance
      .collection('messages')
      .orderBy('timestamp', descending: true)
      .snapshots()
      .map((snap) => snap.docs
          .map((d) => Message.fromJson(d.data()))
          .toList());
});

// Timer/sensor example
final tickProvider = StreamProvider<int>((ref) {
  return Stream.periodic(
    const Duration(seconds: 1),
    (i) => i,
  );
});

// Widget usage is identical to FutureProvider
class ChatScreen extends ConsumerWidget {
  @override
  Widget build(BuildContext context, WidgetRef ref) {
    return ref.watch(messagesProvider).when(
      loading: () => const CircularProgressIndicator(),
      error: (e, s) => Text('Error: $e'),
      data: (messages) => ListView(
        children: messages.map((m) => Text(m.text)).toList(),
      ),
    );
  }
}
09

AsyncNotifierProvider

Combines the power of NotifierProvider and FutureProvider. State is an AsyncValue<T> and you can perform mutations (add, delete, update) while managing loading/error states.

DART 
// ── Code-gen version (recommended) ──────
@riverpod
class UserList extends _$UserList {
  @override
  Future<List<User>> build() async {
    // Fetch on first use
    return ref.watch(userRepositoryProvider).getAll();
  }

  Future<void> addUser(String name) async {
    // Set loading state while preserving old data
    state = const AsyncLoading().copyWithPrevious(state);
    state = await AsyncValue.guard(() async {
      await ref.read(userRepositoryProvider).create(name);
      return ref.read(userRepositoryProvider).getAll();
    });
  }

  Future<void> deleteUser(String id) async {
    state = await AsyncValue.guard(() async {
      await ref.read(userRepositoryProvider).delete(id);
      // Optimistic update: remove from current list
      return state.requireValue
          .where((u) => u.id != id)
          .toList();
    });
  }
}

// Widget Usage
class UsersPage extends ConsumerWidget {
  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final usersAsync = ref.watch(userListProvider);

    return usersAsync.when(
      loading: () => const CircularProgressIndicator(),
      error: (e, _) => Text('Error: $e'),
      data: (users) => Column(children: [
        ElevatedButton(
          onPressed: () => ref.read(userListProvider.notifier).addUser('Alice'),
          child: const Text('Add User'),
        ),
        ...users.map((u) => Text(u.name)),
      ]),
    );
  }
}

Provider Family

Pass parameters to a provider with .family. Each unique parameter creates an independent provider instance.

DART 
// Traditional .family modifier
final userByIdProvider = FutureProvider.family<User, String>(
  (ref, userId) async {
    return ref.watch(userRepositoryProvider).getById(userId);
  },
);

// Code-gen with parameters (preferred)
@riverpod
Future<User> userById(UserByIdRef ref, String userId) async {
  return ref.watch(userRepositoryProvider).getById(userId);
}
// Access: ref.watch(userByIdProvider('user-123'))

// Multiple parameters — use a data class or record
@riverpod
Future<List<Post>> filteredPosts(
  FilteredPostsRef ref, {
  required String userId,
  required String category,
}) async {
  return ref.watch(postRepositoryProvider)
      .getPosts(userId: userId, category: category);
}
// Access: ref.watch(filteredPostsProvider(userId: 'abc', category: 'tech'))

// Usage in widget
class UserProfile extends ConsumerWidget {
  const UserProfile({super.key, required this.userId});
  final String userId;

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    return ref.watch(userByIdProvider(userId)).when(
      loading: () => const CircularProgressIndicator(),
      error: (e, _) => Text('Error: $e'),
      data: (user) => Text(user.name),
    );
  }
}
⚠️ Family parameter equality Parameters must implement == and hashCode. Primitives (String, int, bool) work out-of-the-box. Custom objects need equatable or manual implementation. Use freezed for complex parameter objects.

AutoDispose

When autoDispose is enabled, a provider's state is destroyed when no widget is listening to it. This prevents memory leaks and cache staleness.

DART 
// Explicit autoDispose
final searchResultsProvider = FutureProvider
    .autoDispose
    .family<List<Product>, String>((ref, query) async {
  return ref.watch(productRepositoryProvider).search(query);
});

// Code-gen is autoDispose by default!
@riverpod  // autoDispose by default
Future<List<Product>> searchResults(SearchResultsRef ref, String query) async {
  return ref.watch(productRepositoryProvider).search(query);
}

// Opt out of autoDispose in code-gen
@Riverpod(keepAlive: true)
UserRepository userRepository(UserRepositoryRef ref) {
  return UserRepository();
}

// Keep alive for a duration then dispose
final cachedDataProvider = FutureProvider.autoDispose<Data>((ref) async {
  // Keep alive for 5 minutes even without listeners
  final link = ref.keepAlive();
  Timer(const Duration(minutes: 5), link.close);
  return fetchData();
});

// onDispose — cleanup side effects
final connectionProvider = Provider.autoDispose<WebSocket>((ref) {
  final socket = WebSocket('wss://api.example.com/ws');
  ref.onDispose(() => socket.close());  // ← cleanup
  return socket;
});
✅ Rule of Thumb Use keepAlive: true (or no autoDispose) for global singletons like repositories and services. Use autoDispose (the default in code-gen) for page-specific or feature-specific state.

Combining Providers

Providers can depend on each other using ref.watch() inside the provider body. When a dependency changes, the dependent provider re-evaluates.

DART 
// Filter state
enum Filter { all, active, completed }

final filterProvider = StateProvider<Filter>((ref) => Filter.all);

// All todos (source of truth)
final allTodosProvider = NotifierProvider<TodosNotifier, List<Todo>>(
  TodosNotifier.new,
);

// Filtered todos — depends on BOTH providers
final filteredTodosProvider = Provider<List<Todo>>((ref) {
  final filter = ref.watch(filterProvider);
  final todos = ref.watch(allTodosProvider);

  return switch (filter) {
    Filter.all       => todos,
    Filter.active    => todos.where((t) => !t.completed).toList(),
    Filter.completed => todos.where((t) => t.completed).toList(),
  };
});

// Computed stats derived from todos
final todoStatsProvider = Provider((ref) {
  final todos = ref.watch(allTodosProvider);
  return (
    total: todos.length,
    completed: todos.where((t) => t.completed).length,
    active: todos.where((t) => !t.completed).length,
  );
});

// ── Async dependency chaining ────────────
@riverpod
Future<List<Post>> userPosts(UserPostsRef ref) async {
  // Wait for user to load, then fetch their posts
  final user = await ref.watch(currentUserProvider.future);
  return ref.watch(postRepositoryProvider).getByUser(user.id);
}

// Selective watching — avoid unnecessary rebuilds
final usernameProvider = Provider<String>((ref) {
  // Only rebuild when .name changes, not the whole user
  return ref.watch(userProvider.select((u) => u.name));
});

Testing

Riverpod makes testing easy via ProviderContainer (unit tests) and ProviderScope with overrides (widget tests).

DART 
// ── Unit test with ProviderContainer ────
import 'package:flutter_test/flutter_test.dart';
import 'package:riverpod/riverpod.dart';
import 'package:mocktail/mocktail.dart';

class MockUserRepository extends Mock implements UserRepository {}

void main() {
  group('UserList', () {
    late MockUserRepository mockRepo;

    setUp(() {
      mockRepo = MockUserRepository();
    });

    test('loads users on build', () async {
      when(() => mockRepo.getAll())
          .thenAnswer((_) async => [User(id: '1', name: 'Alice')]);

      final container = ProviderContainer(
        overrides: [
          // Override the repository with a mock
          userRepositoryProvider.overrideWithValue(mockRepo),
        ],
      );
      addTearDown(container.dispose);  // ← always dispose!

      final result = await container.read(userListProvider.future);
      expect(result.first.name, 'Alice');
    });

    test('addUser updates list', () async {
      when(() => mockRepo.getAll()).thenAnswer((_) async => []);
      when(() => mockRepo.create(any())).thenAnswer((_) async => {});

      final container = ProviderContainer(
        overrides: [userRepositoryProvider.overrideWithValue(mockRepo)],
      );
      addTearDown(container.dispose);

      final notifier = container.read(userListProvider.notifier);
      await notifier.addUser('Bob');

      verify(() => mockRepo.create('Bob')).called(1);
    });
  });

  // ── Widget test ──────────────────────────
  testWidgets('UsersPage shows users', (tester) async {
    final mockRepo = MockUserRepository();
    when(() => mockRepo.getAll())
        .thenAnswer((_) async => [User(id: '1', name: 'Alice')]);

    await tester.pumpWidget(
      ProviderScope(
        overrides: [
          userRepositoryProvider.overrideWithValue(mockRepo),
        ],
        child: const MaterialApp(home: UsersPage()),
      ),
    );

    await tester.pumpAndSettle();
    expect(find.text('Alice'), findsOneWidget);
  });
}

// ── domain/user_repository.dart ──────────
abstract interface class UserRepository {
  Future<User> getUser(String id);
}

class UserRepositoryImpl implements UserRepository {
  const UserRepositoryImpl(this._api);
  final UserApi _api;

  @override
  Future<User> getUser(String id) async {
    final dto = await _api.getUser(id);
    return dto.toDomain();  // DTO → Domain model
  }
}

@Riverpod(keepAlive: true)
UserRepository userRepository(UserRepositoryRef ref) =>
    UserRepositoryImpl(ref.watch(userApiProvider));

// ── features/profile/profile_notifier.dart
@riverpod
class ProfileNotifier extends _$ProfileNotifier {
  @override
  Future<User> build(String userId) async {
    return ref.watch(userRepositoryProvider).getUser(userId);
  }

  Future<void> updateName(String newName) async {
    state = const AsyncLoading().copyWithPrevious(state);
    state = await AsyncValue.guard(() async {
      await ref.read(userRepositoryProvider).updateName(state.requireValue.id, newName);
      return state.requireValue.copyWith(name: newName);
    });
  }
}

Performance Tips

Use select() to narrow rebuilds

DART 
// BAD: rebuilds when ANY user field changes
final user = ref.watch(userProvider);
Text(user.name);

// GOOD: only rebuilds when name changes
final name = ref.watch(userProvider.select((u) => u.name));
Text(name);

// Select with AsyncValue
final name = ref.watch(
  userProvider.select((async) => async.whenData((u) => u.name)),
);

// BAD: entire list — rebuilds on any todo change
final count = ref.watch(todosProvider).length;

// GOOD: only rebuilds when count changes
final count = ref.watch(todosProvider.select((todos) => todos.length));

Use Consumer to limit rebuild scope

DART 
// Instead of wrapping the whole page in ConsumerWidget,
// use Consumer() to only rebuild the part that needs it.
class ExpensivePage extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Column(children: [
      const ExpensiveStaticWidget(),  // won't rebuild
      Consumer(
        builder: (context, ref, child) {
          final count = ref.watch(counterProvider);
          return Text('$count');  // only this rebuilds
        },
      ),
    ]);
  }
}
  • Prefer code-gen (@riverpod) — it generates autoDispose by default, preventing memory leaks
  • Use .select() to watch only specific fields of complex objects
  • Avoid calling ref.watch() conditionally — always watch at the top level of build()
  • Use keepAlive: true sparingly — only for expensive-to-create singletons
  • Prefer narrow widgets — wrap only the changing part with Consumer
  • Use ref.listen() for side effects (navigation, dialogs) instead of watchers

class Auth extends _$Auth {
  @override
  Future<AuthState> build() async {
    // Check persisted session on startup
    final token = await SecureStorage.getToken();
    if (token == null) return Unauthenticated();
    try {
      final user = await ref.read(authRepositoryProvider).getMe(token);
      return Authenticated(user);
    } catch (_) {
      await SecureStorage.deleteToken();
      return Unauthenticated();
    }
  }

  Future<void> login(String email, String password) async {
    state = const AsyncData(AuthLoading());
    state = await AsyncValue.guard(() async {
      final response = await ref.read(authRepositoryProvider)
          .login(email, password);
      await SecureStorage.saveToken(response.token);
      return Authenticated(response.user);
    });
  }

  Future<void> logout() async {
    await SecureStorage.deleteToken();
    state = const AsyncData(Unauthenticated());
  }
}

Pagination

@riverpod
class PostList extends _$PostList {
  static const _pageSize = 20;
  var _page = 0;
  var _hasMore = true;

  @override
  Future<List<Post>> build() async {
    _page = 0;
    _hasMore = true;
    return _fetchPage(0);
  }

  Future<List<Post>> _fetchPage(int page) async {
    return ref.watch(postRepositoryProvider)
        .getPosts(page: page, limit: _pageSize);
  }

  Future<void> fetchMore() async {
    if (!_hasMore) return;
    final current = state.requireValue;
    state = const AsyncLoading().copyWithPrevious(AsyncData(current));

    final next = await _fetchPage(++_page);
    if (next.length < _pageSize) _hasMore = false;
    state = AsyncData([...current, ...next]);
  }

  bool get hasMore => _hasMore;
}

// ── Infinite scroll widget ─────────────
class PostListView extends ConsumerStatefulWidget {
  @override
  ConsumerState<PostListView> createState() => _PostListViewState();
}

class _PostListViewState extends ConsumerState<PostListView> {
  late final ScrollController _scrollCtrl;

  @override
  void initState() {
    super.initState();
    _scrollCtrl = ScrollController()
      ..addListener(() {
        final pos = _scrollCtrl.position;
        if (pos.pixels >= pos.maxScrollExtent - 200) {
          ref.read(postListProvider.notifier).fetchMore();
        }
      });
  }

  @override
  Widget build(BuildContext context) {
    return ref.watch(postListProvider).when(
      loading: () => const CircularProgressIndicator(),
      error: (e, s) => Text('Error: $e'),
      data: (posts) => ListView.builder(
        controller: _scrollCtrl,
        itemCount: posts.length + 1,
        itemBuilder: (ctx, i) {
          if (i == posts.length) {
            return ref.watch(postListProvider.notifier).hasMore
                ? const CircularProgressIndicator()
                : const Text('No more posts');
          }
          return Text(posts[i].title);
        },
      ),
    );
  }
}

// Cache-first strategy with Hive
@riverpod
class CachedProducts extends _$CachedProducts {
  @override
  Future<List<Product>> build() async {
    final cache = ref.watch(localCacheProvider);
    final api = ref.watch(productApiProvider);

    // Return cached immediately
    final cached = await cache.getProducts();
    if (cached.isNotEmpty) {
      // Refresh in background
      _refreshInBackground(api, cache);
      return cached;
    }

    // No cache → fetch and store
    final products = await api.getProducts();
    await cache.saveProducts(products);
    return products;
  }

  void _refreshInBackground(ProductApi api, LocalCache cache) {
    Future.microtask(() async {
      try {
        final fresh = await api.getProducts();
        await cache.saveProducts(fresh);
        state = AsyncData(fresh);
      } catch (_) { /* silently fail */ }
    });
  }
}
